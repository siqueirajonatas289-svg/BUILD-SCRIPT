-- Função de animação de flutuação e giro suave
local function animatePart(part, baseCFrame, index)
    local time = tick()
    local floatOffset = math.sin(time * 2 + index) * 0.5
    local rotationOffset = CFrame.Angles(0, math.sin(time) * 0.1, 0)
    return baseCFrame * CFrame.new(0, floatOffset, 0) * rotationOffset
end

-- Loop Anti-Sleep e Força Bruta de Network Ownership
task.spawn(function()
    while task.wait(0.1) do -- Intervalo reduzido para manter a posse constante
        if vacuumActive then
            pcall(function()
                player.SimulationRadius = math.huge
                if sethiddenproperty then
                    sethiddenproperty(player, "SimulationRadius", 1e6)
                    sethiddenproperty(player, "MaxSimulationRadius", 1e6)
                end
            end)
            
            -- Aplicar micro-movimentos para enganar o motor físico (Evita que caiam)
            for part, _ in pairs(animatedParts) do
                if part and part.Parent then
                    -- Uma velocidade insignificante visualmente, mas que impede o status "Sleeping"
                    part.AssemblyLinearVelocity = Vector3.new(0, 0.01, 0)
                    part.AssemblyAngularVelocity = Vector3.new(0.01, 0, 0.01)
                end
            end
        end
    end
end)

    if vacuumActive then
        local timeTick = tick() * 0.5 
        local heightOffset = math.sin(timeTick) * 15 
        
        for part, basePos in pairs(elevatorParts) do
            if part and part:FindFirstChild("VacuumForce") then
                part.VacuumForce.Position = basePos + Vector3.new(0, heightOffset, 0)
            end
        end

        local activeOrigin = fixedOrigin
        if followCharacter and targetMode == "Character" then
            local char = player.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            if root then
                activeOrigin = root.Position
            end
        end

        for part, data in pairs(animatedParts) do
            if part and part:FindFirstChild("VacuumForce") then
                part.VacuumForce.Position = activeOrigin + data.relativePos + Vector3.new(0, math.sin(tick() * 3 + data.index) * 0.5, 0)
                part.VacuumGyro.CFrame = CFrame.Angles(0, math.sin(tick()) * 0.2, 0)
            end
        end
    end
end)

-- Lógica Geométrica Avançada
local function generateSpherePoints(count, radius, offsetX, offsetY, offsetZ)
    local points = {}
    local phi = math.pi * (3 - math.sqrt(5)) -- Proporção áurea
    for i = 0, count - 1 do
        local y = 1 - (i / (count - 1)) * 2
        local r = math.sqrt(1 - y * y)
        local theta = phi * i
        local x = math.cos(theta) * r
        local z = math.sin(theta) * r
        table.insert(points, Vector3.new(x * radius + offsetX, y * radius + offsetY, z * radius + offsetZ))
    end
    return points
end

local function generateCylinderPoints(count, radius, height, offsetX, offsetY, offsetZ)
    local points = {}
    local layers = math.floor(math.sqrt(count))
    local perLayer = math.ceil(count / layers)
    for i = 0, count - 1 do
        local layer = math.floor(i / perLayer)
        local posInLayer = i % perLayer
        local angle = (posInLayer / perLayer) * math.pi * 2
        local h = (layer / math.max(1, layers - 1)) * height
        
        -- Adiciona um espiralamento interno para preencher o cilindro
        local r = radius * (0.3 + 0.7 * (posInLayer % 3) / 2) 
        
        local x = math.cos(angle) * r
        local z = math.sin(angle) * r
        table.insert(points, Vector3.new(x + offsetX, h + offsetY, z + offsetZ))
    end
    return points
end

-- Função para atualizar construções ao vivo
local function updatePositions()
    if not vacuumActive then return end
    
    local currentCharacter = player.Character
    local root = currentCharacter and currentCharacter:FindFirstChild("HumanoidRootPart")
    if not root and targetMode == "Character" then return end
    local originPosition = (targetMode == "MapCenter") and Vector3.new(0, 45, 0) or root.Position
    fixedOrigin = originPosition
    
    local objects = {}
    for _, part in pairs(workspace:GetDescendants()) do
        if part:IsA("BasePart") and part:FindFirstChild("VacuumForce") then
            table.insert(objects, part)
        end
    end
    table.sort(objects, function(a, b) return a:GetMass() > b:GetMass() end)
    
    local totalObjects = #objects
    if totalObjects == 0 then return end
    
    animatedParts = {}

    -- Pre-calcular posições para Troll3 para melhor densidade
    local troll3Points = {}
    if structureMode == "Troll3" then
        local ballCount = math.floor(totalObjects * 0.25)
        local shaftCount = totalObjects - (ballCount * 2)
        local ballRadius = bWidth / 3
        
        local leftBall = generateSpherePoints(ballCount, ballRadius, -ballRadius * 1.2, ballRadius, 0)
        local rightBall = generateSpherePoints(ballCount, ballRadius, ballRadius * 1.2, ballRadius, 0)
        local shaft = generateCylinderPoints(shaftCount, ballRadius * 0.8, bHeight, 0, ballRadius * 1.5, 0)
        
        for _, p in ipairs(leftBall) do table.insert(troll3Points, p) end
        for _, p in ipairs(rightBall) do table.insert(troll3Points, p) end
        for _, p in ipairs(shaft) do table.insert(troll3Points, p) end
    end

    for i, part in ipairs(objects) do
        local targetPos = originPosition
        
        if structureMode == "Building" then
            -- Paredes densas formando um cuboide reto
            local perFloor = math.floor(totalObjects / math.max(1, (bHeight / 4)))
            local floor = math.floor(i / perFloor)
            local perimeterPos = (i % perFloor) / perFloor
            local x, z = 0, 0
            if perimeterPos < 0.25 then x = -bWidth/2 + (perimeterPos*4)*bWidth; z = -bWidth/2
            elseif perimeterPos < 0.5 then x = bWidth/2; z = -bWidth/2 + ((perimeterPos-0.25)*4)*bWidth
            elseif perimeterPos < 0.75 then x = bWidth/2 - ((perimeterPos-0.5)*4)*bWidth; z = bWidth/2
            else x = -bWidth/2; z = bWidth/2 - ((perimeterPos-0.75)*4)*bWidth end
            targetPos = originPosition + Vector3.new(x, floor * 4, z)
            
        elseif structureMode == "Restaurant" then
            -- Base densa, paredes laterais baixas e teto largo
            local baseCount = math.floor(totalObjects * 0.4)
            local wallCount = math.floor(totalObjects * 0.3)
            local roofCount = totalObjects - baseCount - wallCount
            
            if i <= baseCount then
                local row = math.ceil(math.sqrt(baseCount))
                local x = (i % row) * (bWidth / row) - (bWidth / 2)
                local z = math.floor(i / row) * (bWidth / row) - (bWidth / 2)
                targetPos = originPosition + Vector3.new(x, 0, z)
            elseif i <= baseCount + wallCount then
                local idx = i - baseCount
                local perimeterPos = (idx / wallCount)
                local wallHeight = 12
                local h = (idx % 3) * (wallHeight / 3)
                local x, z = 0, 0
                if perimeterPos < 0.25 then x = -bWidth/2 + (perimeterPos*4)*bWidth; z = -bWidth/2
                elseif perimeterPos < 0.5 then x = bWidth/2; z = -bWidth/2 + ((perimeterPos-0.25)*4)*bWidth
                elseif perimeterPos < 0.75 then x = bWidth/2 - ((perimeterPos-0.5)*4)*bWidth; z = bWidth/2
                else x = -bWidth/2; z = bWidth/2 - ((perimeterPos-0.75)*4)*bWidth end
                targetPos = originPosition + Vector3.new(x, h, z)
            else
                local idx = i - (baseCount + wallCount)
                local row = math.ceil(math.sqrt(roofCount))
                local roofWidth = bWidth * 1.2 -- Teto com borda maior
                local x = (idx % row) * (roofWidth / row) - (roofWidth / 2)
                local z = math.floor(idx / row) * (roofWidth / row) - (roofWidth / 2)
                targetPos = originPosition + Vector3.new(x, 12, z)
            end
            
        elseif structureMode == "Cage" then
            local offset = bWidth/2
            if i % 3 == 0 then targetPos = originPosition + Vector3.new(offset, (i/totalObjects)*bHeight, 0)
            elseif i % 3 == 1 then targetPos = originPosition + Vector3.new(-offset, (i/totalObjects)*bHeight, 0)
            else targetPos = originPosition + Vector3.new(0, (i/totalObjects)*bHeight, offset) end
            
        elseif structureMode == "Troll3" then
            if troll3Points[i] then
                targetPos = originPosition + troll3Points[i]
            end
            
        elseif structureMode == "Pyramid" then
            local height = (i / totalObjects) * bHeight; local radius = (1 - (i / totalObjects)) * (bWidth / 2)
            targetPos = originPosition + Vector3.new(math.sin(i * 0.5) * radius, height, math.cos(i * 0.5) * radius)
        elseif structureMode == "Helix" then
            targetPos = originPosition + Vector3.new(math.sin(i * 0.3) * (bWidth / 4), (i / totalObjects) * bHeight, math.cos(i * 0.3) * (bWidth / 4))
        elseif structureMode == "Sphere" then
            local r = bWidth / 2; local phi = math.acos(-1 + 2 * i / totalObjects); local theta = math.sqrt(totalObjects * math.pi) * phi
            targetPos = originPosition + Vector3.new(r * math.cos(theta) * math.sin(phi), r * math.cos(phi), r * math.sin(theta) * math.sin(phi))
        elseif structureMode == "Ring" then
            local angle = (i / totalObjects) * math.pi * 2
            targetPos = originPosition + Vector3.new(math.sin(angle) * bWidth, 0, math.cos(angle) * bWidth)
        elseif structureMode == "Stairs" then
            targetPos = originPosition + Vector3.new((i % 5) * 2, (math.floor(i / 5) * 3), (math.floor(i / 5) * 2))
        elseif structureMode == "Cone" then
            local h = (i / totalObjects) * bHeight; local r = (1 - (i / totalObjects)) * (bWidth / 2)
            targetPos = originPosition + Vector3.new(math.sin(i * 0.5) * r, h, math.cos(i * 0.5) * r)
        elseif structureMode == "Cross" then
            local dir = i % 4; local dist = (i / totalObjects) * bWidth
            if dir == 0 then targetPos = originPosition + Vector3.new(dist, 0, 0)
            elseif dir == 1 then targetPos = originPosition + Vector3.new(-dist, 0, 0)
            elseif dir == 2 then targetPos = originPosition + Vector3.new(0, 0, dist)
            else targetPos = originPosition + Vector3.new(0, 0, -dist) end
        elseif structureMode == "Wall" then
            targetPos = originPosition + Vector3.new(((i % 20) * 3) - 30, 0, math.floor(i / 20) * 3)
        elseif structureMode == "Cube" then
            local edge = i % 4; local pos = ((i / totalObjects) * bWidth) - (bWidth / 2)
            if edge == 0 then targetPos = originPosition + Vector3.new(pos, -bWidth/2, -bWidth/2)
            elseif edge == 1 then targetPos = originPosition + Vector3.new(pos, bWidth/2, bWidth/2)
            elseif edge == 2 then targetPos = originPosition + Vector3.new(-bWidth/2, pos, -bWidth/2)
            else targetPos = originPosition + Vector3.new(bWidth/2, pos, bWidth/2) end
        end
        
        animatedParts[part] = {relativePos = targetPos - originPosition, index = i}
        
        -- Atualizar imediatamente a posição âncora principal da peça
        if part:FindFirstChild("VacuumForce") then
             part.VacuumForce.Position = targetPos
        end
    end
end

    minus.Size = UDim2.new(0.2, 0, 1, 0)
    minus.Position = UDim2.new(0.4, 0, 0, 0)
    minus.Text = "-"

    local valLbl = Instance.new("TextLabel", frame)
    valLbl.Size = UDim2.new(0.2, 0, 1, 0)
    valLbl.Position = UDim2.new(0.6, 0, 0, 0)
    valLbl.Text = tostring(initialValue)
    valLbl.TextColor3 = Color3.fromRGB(40, 200, 40)
    valLbl.BackgroundTransparency = 1
    
    local plus = Instance.new("TextButton", frame)
    plus.Size = UDim2.new(0.2, 0, 1, 0)
    plus.Position = UDim2.new(0.8, 0, 0, 0)
    plus.Text = "+"
    plus.BackgroundColor3 = Color3.fromRGB(50,50,50)
    plus.TextColor3 = Color3.new(1,1,1)
    
    minus.MouseButton1Click:Connect(function()
        callback(-2)
        valLbl.Text = tostring(callback(0))
        updatePositions()
    end)
    plus.MouseButton1Click:Connect(function()
        callback(2)
        valLbl.Text = tostring(callback(0))
        updatePositions()
    end)
    
    return frame
end

local function isFreePhysicsPart(part, currentCharacter)
    if not part:IsA("BasePart") or part.Anchored then return false end
    if currentCharacter and part:IsDescendantOf(currentCharacter) then return false end
    local ancestor = part.Parent
    while ancestor and ancestor ~= workspace do
        if ancestor:FindFirstChildOfClass("Humanoid") then return false end
        ancestor = ancestor.Parent
    end
    return true
end

local function clearVacuumForces()
    elevatorParts = {}
    animatedParts = {}
    for _, part in pairs(workspace:GetDescendants()) do
        if part:IsA("BasePart") then
            local f = part:FindFirstChild("VacuumForce") local g = part:FindFirstChild("VacuumGyro")
            if f then f:Destroy() end if g then g:Destroy() end
            pcall(function() part.AssemblyLinearVelocity = Vector3.zero; part.AssemblyAngularVelocity = Vector3.zero end)
        end
    end
end

activateBtn.MouseButton1Click:Connect(function()
    local currentCharacter = player.Character
    if vacuumActive then
        vacuumActive = false activateBtn.Text = "ENABLE VACUUM" activateBtn.BackgroundColor3 = Color3.fromRGB(40, 200, 40) clearVacuumForces() return
    end
    local root = currentCharacter and currentCharacter:FindFirstChild("HumanoidRootPart")
    if not root and targetMode == "Character" then return end
    vacuumActive = true activateBtn.Text = "DISABLE VACUUM" activateBtn.BackgroundColor3 = Color3.fromRGB(200, 40, 40)
    
    local originPosition = (targetMode == "MapCenter") and Vector3.new(0, 45, 0) or root.Position
    local radius = tonumber(radiusInput.Text) or 150
    
    local objects = {}
    for _, p in pairs(workspace:GetDescendants()) do
        if isFreePhysicsPart(p, currentCharacter) and (p.Position - originPosition).Magnitude <= radius then
            p.CanCollide = false
            pcall(function() p.CustomPhysicalProperties = PhysicalProperties.new(0,0,0,0,0) end)
            for _, child in pairs(p:GetChildren()) do if child.Name:find("Vacuum") then child:Destroy() end end
            
            local bp = Instance.new("BodyPosition") bp.Name = "VacuumForce" bp.MaxForce = Vector3.new(math.huge, math.huge, math.huge) bp.P = 20000 bp.D = 800 bp.Parent = p
            local bg = Instance.new("BodyGyro") bg.Name = "VacuumGyro" bg.MaxTorque = Vector3.new(math.huge, math.huge, math.huge) bg.Parent = p
            table.insert(objects, p)
        end
    end
    updatePositions()
end)

-- ==========================================
-- SISTEMA DE ANIMAÇÃO DE INICIALIZAÇÃO
-- ==========================================
task.spawn(function()
    -- 1. Atraso inicial de 3 segundos exigido pelo usuário
    task.wait(3)
    
    -- 2. Fade In do Fundo da Intro
    local bgTween = TweenService:Create(introFrame, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.1})
    bgTween:Play()
    bgTween.Completed:Wait()

    -- 3. Animação de Entrada e Movimentação dos Créditos (UI se mexendo)
    introTitle.Rotation = -10
    introCreator.Rotation = 10
    
    local titleTween = TweenService:Create(introTitle, TweenInfo.new(1.5, Enum.EasingStyle.Elastic, Enum.EasingDirection.Out), {
        Position = UDim2.new(0.5, -300, 0.4, -40),
        TextTransparency = 0,
        Rotation = 0
    })
    
    local creatorTween = TweenService:Create(introCreator, TweenInfo.new(1.5, Enum.EasingStyle.Elastic, Enum.EasingDirection.Out), {
        Position = UDim2.new(0.5, -200, 0.5, 20),
        TextTransparency = 0,
        Rotation = 0
    })
    
    titleTween:Play()
    creatorTween:Play()
    task.wait(2.5) -- Tempo para exibição dos créditos

    -- 4. Fade Out da Intro
    TweenService:Create(introTitle, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {TextTransparency = 1, Position = UDim2.new(0.5, -300, 0.3, -40)}):Play()
    TweenService:Create(introCreator, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {TextTransparency = 1, Position = UDim2.new(0.5, -200, 0.6, 20)}):Play()
    task.wait(0.5)
    TweenService:Create(introFrame, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {BackgroundTransparency = 1}):Play()
    task.wait(0.5)
    introFrame:Destroy()

    -- 5. Preparação da MainFrame para animação Canto Esquerdo Acima -> Canto Direito Abaixo
    mainFrame.Size = UDim2.new(0, 0, 0, 0)
    mainFrame.Position = UDim2.new(0, 0, 0, 0) -- Origem: Canto superior esquerdo absoluto
    mainFrame.ClipsDescendants = true
    mainFrame.Visible = true

    -- Mapear os elementos originais e colocá-los invisíveis e deslocados para animação de surgimento
    local animatableElements = {}
    for _, obj in pairs(contentFrame:GetDescendants()) do
        if obj:IsA("GuiObject") and not obj:IsA("UICorner") then
            local data = {
                instance = obj,
                originalPos = obj.Position,
                originalBgTrans = obj.BackgroundTransparency,
                originalTextTrans = (obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox")) and obj.TextTransparency or nil
            }
            table.insert(animatableElements, data)
            
            -- Aplicar o deslocamento inicial e transparência
            obj.Position = obj.Position + UDim2.new(0, 40, 0, 40)
            obj.BackgroundTransparency = 1
            if data.originalTextTrans then obj.TextTransparency = 1 end
        end
    end

    -- 6. Expansão diagonal da MainFrame (Canto esquerdo superior para sua posição final no centro/inferior)
    local expandTween = TweenService:Create(mainFrame, TweenInfo.new(1.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
        Size = UDim2.new(0, 600, 0, 220),
        Position = UDim2.new(0.5, -300, 0.5, -110)
    })
    expandTween:Play()
    expandTween.Completed:Wait()
    mainFrame.ClipsDescendants = false

    -- 7. Loop para revelar cada item da interface de forma única em cascata
    for i, data in ipairs(animatableElements) do
        local obj = data.instance
        task.delay(i * 0.05, function()
            local props = {Position = data.originalPos, BackgroundTransparency = data.originalBgTrans}
            if data.originalTextTrans then props.TextTransparency = data.originalTextTrans end
            
            -- Animação tipo Back para dar um efeito de "pop-up" em cada elemento
            TweenService:Create(obj, TweenInfo.new(0.6, Enum.EasingStyle.Back, Enum.EasingDirection.Out), props):Play()
        end)
    end
end)
